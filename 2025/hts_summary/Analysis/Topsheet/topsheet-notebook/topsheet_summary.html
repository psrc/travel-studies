## persons

### work location

```{r}
#| out-width: "60%"
#| fig-align: "center"

df_sum <- psrc_hts_stat(df_hts_analysis, "person", c("telecommute_status"), incl_na = FALSE) %>%
  get_percent_format() %>%
  group_by(survey_year) %>%
  mutate(label_ypos=cumsum(prop))

ggplot(df_sum, aes(x=factor(survey_year), y=prop, fill=fct_rev(telecommute_status))) +
  geom_bar(stat="identity",width = 0.8) +
  geom_text(aes(y=label_ypos,label=prop_label), 
            vjust = 1.5,
            color="white") +
  scale_fill_manual(values = c("Fully At Home"="#F05A28",
                               "Hybrid"="#91268F",
                               "Fully In Person"="#76787A"))+
  scale_y_continuous(labels = scales::percent) +
  psrc_style() +
  theme(panel.grid.major.y = element_blank(),
        axis.title = element_blank())

```
